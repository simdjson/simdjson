110// Our fast kernels require 64-bit systems.
111//
112// On 32-bit x86, we lack 64-bit popcnt, lzcnt, blsr instructions.
113// Furthermore, the number of SIMD registers is reduced.
114//
115// On 32-bit ARM, we would have smaller registers.
116//
117// The simdjson users should still have the fallback kernel. It is
118// slower, but it should run everywhere.
119
120//
121// Enable valid runtime implementations, and select SIMDJSON_BUILTIN_IMPLEMENTATION
122//
123
124// We are going to use runtime dispatch.
125#if defined(SIMDJSON_IS_X86_64) || defined(SIMDJSON_IS_LSX)
126#ifdef __clang__
127// clang does not have GCC push pop
128// warning: clang attribute push can't be used within a namespace in clang up
129// til 8.0 so SIMDJSON_TARGET_REGION and SIMDJSON_UNTARGET_REGION must be *outside* of a
130// namespace.
131#define SIMDJSON_TARGET_REGION(T) \
132 _Pragma(SIMDJSON_STRINGIFY( \
133 clang attribute push(__attribute__((target(T))), apply_to = function)))
134#define SIMDJSON_UNTARGET_REGION _Pragma("clang attribute pop")
135#elif defined(__GNUC__)
136// GCC is easier
137#define SIMDJSON_TARGET_REGION(T) \
138 _Pragma("GCC push_options") _Pragma(SIMDJSON_STRINGIFY(GCC target(T)))
139#define SIMDJSON_UNTARGET_REGION _Pragma("GCC pop_options")
140#endif // clang then gcc
141
142#endif // defined(SIMDJSON_IS_X86_64) || defined(SIMDJSON_IS_LSX)
143
144// Default target region macros don't do anything.
145#ifndef SIMDJSON_TARGET_REGION
146#define SIMDJSON_TARGET_REGION(T)
