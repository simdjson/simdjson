79simple_needs_escaping(std::string_view v) {

90simdjson_inline bool fast_needs_escaping(std::string_view view) {
91 if (view.size() < 16) {
92 return simple_needs_escaping(view);
93 }
94 size_t i = 0;
95 uint8x16_t running = vdupq_n_u8(0);
96 uint8x16_t v34 = vdupq_n_u8(34);
97 uint8x16_t v92 = vdupq_n_u8(92);
98
99 for (; i + 15 < view.size(); i += 16) {
100 uint8x16_t word = vld1q_u8((const uint8_t *)view.data() + i);
101 running = vorrq_u8(running, vceqq_u8(word, v34));
102 running = vorrq_u8(running, vceqq_u8(word, v92));
103 running = vorrq_u8(running, vcltq_u8(word, vdupq_n_u8(32)));
104 }
105 if (i < view.size()) {
106 uint8x16_t word =
107 vld1q_u8((const uint8_t *)view.data() + view.length() - 16);
108 running = vorrq_u8(running, vceqq_u8(word, v34));
109 running = vorrq_u8(running, vceqq_u8(word, v92));
110 running = vorrq_u8(running, vcltq_u8(word, vdupq_n_u8(32)));
111 }
112 return vmaxvq_u32(vreinterpretq_u32_u8(running)) != 0;
113}

150find_next_json_quotable_character_scalar(const std::string_view view,
151 size_t location) noexcept {
152 for (auto pos = view.begin() + location; pos != view.end(); ++pos) {

160// SIMD-accelerated position finding that directly locates the first quotable
161// character, combining detection and position extraction in a single pass to
162// minimize redundant work.
163#if SIMDJSON_EXPERIMENTAL_HAS_NEON
164simdjson_inline size_t
165find_next_json_quotable_character(const std::string_view view,
166 size_t location) noexcept {
167 const size_t len = view.size();
168 const uint8_t *ptr =
169 reinterpret_cast<const uint8_t *>(view.data()) + location;
170 size_t remaining = len - location;
171
172 // SIMD constants for characters requiring escape
173 uint8x16_t v34 = vdupq_n_u8(34); // '"'
174 uint8x16_t v92 = vdupq_n_u8(92); // '\\'
175 uint8x16_t v32 = vdupq_n_u8(32); // control char threshold
176
177 while (remaining >= 16) {
178 uint8x16_t word = vld1q_u8(ptr);
179
180 // Check for quotable characters: '"', '\\', or control chars (< 32)
181 uint8x16_t needs_escape = vceqq_u8(word, v34);
182 needs_escape = vorrq_u8(needs_escape, vceqq_u8(word, v92));
183 needs_escape = vorrq_u8(needs_escape, vcltq_u8(word, v32));
184
185 if (vmaxvq_u32(vreinterpretq_u32_u8(needs_escape)) != 0) {
186 // Found quotable character - extract exact byte position using ctz
187 uint64x2_t as64 = vreinterpretq_u64_u8(needs_escape);
188 uint64_t lo = vgetq_lane_u64(as64, 0);
189 uint64_t hi = vgetq_lane_u64(as64, 1);
190 size_t offset = ptr - reinterpret_cast<const uint8_t *>(view.data());
191#ifdef _MSC_VER
192 unsigned long trailing_zero = 0;
193 if (lo != 0) {
194 _BitScanForward64(&trailing_zero, lo);
195 return offset + trailing_zero / 8;
196 } else {
197 _BitScanForward64(&trailing_zero, hi);
198 return offset + 8 + trailing_zero / 8;
199 }
200#else
201 if (lo != 0) {
202 return offset + __builtin_ctzll(lo) / 8;
203 } else {
204 return offset + 8 + __builtin_ctzll(hi) / 8;
205 }
206#endif
207 }
208 ptr += 16;
209 remaining -= 16;
210 }
211
212 // Scalar fallback for remaining bytes
213 size_t current = len - remaining;
214 return find_next_json_quotable_character_scalar(view, current);
215}
216#elif SIMDJSON_EXPERIMENTAL_HAS_SSE2
217simdjson_inline size_t
218find_next_json_quotable_character(const std::string_view view,
219 size_t location) noexcept {
220 const size_t len = view.size();
221 const uint8_t *ptr =
222 reinterpret_cast<const uint8_t *>(view.data()) + location;
223 size_t remaining = len - location;
224
225 // SIMD constants
226 __m128i v34 = _mm_set1_epi8(34); // '"'
227 __m128i v92 = _mm_set1_epi8(92); // '\\'
228 __m128i v31 = _mm_set1_epi8(31); // for control char detection
229
230 while (remaining >= 16) {
231 __m128i word = _mm_loadu_si128(reinterpret_cast<const __m128i *>(ptr));
232
233 // Check for quotable characters
234 __m128i needs_escape = _mm_cmpeq_epi8(word, v34);
235 needs_escape = _mm_or_si128(needs_escape, _mm_cmpeq_epi8(word, v92));
236 needs_escape = _mm_or_si128(
237 needs_escape,
238 _mm_cmpeq_epi8(_mm_subs_epu8(word, v31), _mm_setzero_si128()));
239
240 int mask = _mm_movemask_epi8(needs_escape);
241 if (mask != 0) {
242 // Found quotable character - use trailing zero count to find position
243 size_t offset = ptr - reinterpret_cast<const uint8_t *>(view.data());
244#ifdef _MSC_VER
245 unsigned long trailing_zero = 0;
246 _BitScanForward(&trailing_zero, mask);
247 return offset + trailing_zero;
248#else
249 return offset + __builtin_ctz(mask);
250#endif
251 }
252 ptr += 16;
253 remaining -= 16;
254 }
255
256 // Scalar fallback for remaining bytes
257 size_t current = len - remaining;
258 return find_next_json_quotable_character_scalar(view, current);
259}

262find_next_json_quotable_character(const std::string_view view,
263 size_t location) noexcept {
264 return find_next_json_quotable_character_scalar(view, location);
265}

268SIMDJSON_CONSTEXPR_LAMBDA static std::string_view control_chars[] = {
269 "\\u0000", "\\u0001", "\\u0002", "\\u0003", "\\u0004", "\\u0005", "\\u0006",
270 "\\u0007", "\\b", "\\t", "\\n", "\\u000b", "\\f", "\\r",
271 "\\u000e", "\\u000f", "\\u0010", "\\u0011", "\\u0012", "\\u0013", "\\u0014",
272 "\\u0015", "\\u0016", "\\u0017", "\\u0018", "\\u0019", "\\u001a", "\\u001b",
273 "\\u001c", "\\u001d", "\\u001e", "\\u001f"};

275// All Unicode characters may be placed within the quotation marks, except for
276// the characters that MUST be escaped: quotation mark, reverse solidus, and the
277// control characters (U+0000 through U+001F). There are two-character sequence
278// escape representations of some popular characters:
279// \", \\, \b, \f, \n, \r, \t.
280SIMDJSON_CONSTEXPR_LAMBDA simdjson_inline void escape_json_char(char c, char *&out) {
281 if (c == '"') {
282 memcpy(out, "\\\"", 2);
283 out += 2;
284 } else if (c == '\\') {
285 memcpy(out, "\\\\", 2);
286 out += 2;
287 } else {
288 std::string_view v = control_chars[uint8_t(c)];
289 memcpy(out, v.data(), v.size());
290 out += v.size();
291 }
292}

294// Writes the escaped version of input to out, returning the number of bytes
295// written. Uses SIMD position finding to locate quotable characters efficiently.
296inline size_t write_string_escaped(const std::string_view input, char *out) {
297 size_t mysize = input.size();
298
299 // Use SIMD position finder directly - it returns mysize if no escape needed
300 size_t location = find_next_json_quotable_character(input, 0);
301 if (location == mysize) {
302 // Fast path: no escaping needed
303 memcpy(out, input.data(), input.size());
304 return input.size();
305 }
306
307 const char *const initout = out;
308 memcpy(out, input.data(), location);
309 out += location;
310 escape_json_char(input[location], out);
311 location += 1;
312 while (location < mysize) {
313 size_t newlocation = find_next_json_quotable_character(input, location);
314 memcpy(out, input.data() + location, newlocation - location);
315 out += newlocation - location;
316 location = newlocation;
317 if (location == mysize) {
318 break;
319 }
320 escape_json_char(input[location], out);
321 location += 1;
322 }
323 return out - initout;
324}

331simdjson_inline bool string_builder::capacity_check(size_t upcoming_bytes) {
332 // We use the convention that when is_valid is false, then the capacity and
333 // the position are 0.
334 // Most of the time, this function will return true.
335 if (simdjson_likely(upcoming_bytes <= capacity - position)) {
336 return true;
337 }
338 // check for overflow, most of the time there is no overflow
339 if (simdjson_likely(position + upcoming_bytes < position)) {
340 return false;
341 }
342 // We will rarely get here.
343 grow_buffer((std::max)(capacity * 2, position + upcoming_bytes));
344 // If the buffer allocation failed, we set is_valid to false.
345 return is_valid;
346}
347
348simdjson_inline void string_builder::grow_buffer(size_t desired_capacity) {
349 if (!is_valid) {
350 return;
351 }
352 std::unique_ptr<char[]> new_buffer(new (std::nothrow) char[desired_capacity]);
353 if (new_buffer.get() == nullptr) {
354 set_valid(false);
355 return;
356 }
357 std::memcpy(new_buffer.get(), buffer.get(), position);
358 buffer.swap(new_buffer);
359 capacity = desired_capacity;
360}

367 buffer.reset();

369 is_valid = true;
370 }
371}

481template <typename number_type, typename>
482simdjson_inline void string_builder::append(number_type v) noexcept {
483 static_assert(std::is_same<number_type, bool>::value ||
484 std::is_integral<number_type>::value ||
485 std::is_floating_point<number_type>::value,
486 "Unsupported number type");
487 // If C++17 is available, we can 'if constexpr' here.
488 SIMDJSON_IF_CONSTEXPR(std::is_same<number_type, bool>::value) {
489 if (v) {
490 constexpr char true_literal[] = "true";
491 constexpr size_t true_len = sizeof(true_literal) - 1;
492 if (capacity_check(true_len)) {
493 std::memcpy(buffer.get() + position, true_literal, true_len);
494 position += true_len;
495 }
496 } else {
497 constexpr char false_literal[] = "false";
498 constexpr size_t false_len = sizeof(false_literal) - 1;
499 if (capacity_check(false_len)) {
500 std::memcpy(buffer.get() + position, false_literal, false_len);
501 position += false_len;
502 }
503 }
504 }
505 else SIMDJSON_IF_CONSTEXPR(std::is_unsigned<number_type>::value) {
506 // Process 4 digits at a time instead of 2, reducing store operations
507 // and divisions by approximately half for large numbers.
508 constexpr size_t max_number_size = 20;
509 if (capacity_check(max_number_size)) {
510 using unsigned_type = typename std::make_unsigned<number_type>::type;
511 unsigned_type pv = static_cast<unsigned_type>(v);
512 size_t dc = internal::digit_count(pv);
513 char *write_pointer = buffer.get() + position + dc - 1;
514
515 // Process 4 digits per iteration for large numbers
516 while (pv >= 10000) {
517 unsigned_type q = pv / 10000;
518 unsigned_type r = pv % 10000;
519 unsigned_type r_hi = r / 100; // High 2 digits of remainder
520 unsigned_type r_lo = r % 100; // Low 2 digits of remainder
521 // Write low 2 digits first (rightmost), then high 2 digits
522 memcpy(write_pointer - 1, &internal::decimal_table[r_lo * 2], 2);
523 memcpy(write_pointer - 3, &internal::decimal_table[r_hi * 2], 2);
524 write_pointer -= 4;
525 pv = q;
526 }
527
528 // Handle remaining 1-4 digits with original 2-digit loop
529 while (pv >= 100) {
530 memcpy(write_pointer - 1, &internal::decimal_table[(pv % 100) * 2], 2);
531 write_pointer -= 2;
532 pv /= 100;
533 }
534 if (pv >= 10) {
535 *write_pointer-- = char('0' + (pv % 10));
536 pv /= 10;
537 }
538 *write_pointer = char('0' + pv);
539 position += dc;
540 }
541 }
542 else SIMDJSON_IF_CONSTEXPR(std::is_integral<number_type>::value) {
543 // Same 4-digit batching as unsigned path for signed integers
544 constexpr size_t max_number_size = 20;
545 if (capacity_check(max_number_size)) {
546 using unsigned_type = typename std::make_unsigned<number_type>::type;
547 bool negative = v < 0;
548 unsigned_type pv = static_cast<unsigned_type>(v);
549 if (negative) {
550 pv = 0 - pv; // the 0 is for Microsoft
551 }
552 size_t dc = internal::digit_count(pv);
553 // by always writing the minus sign, we avoid the branch.
554 buffer.get()[position] = '-';
555 position += negative ? 1 : 0;
556 char *write_pointer = buffer.get() + position + dc - 1;
557
558 // Process 4 digits per iteration for large numbers
559 while (pv >= 10000) {
560 unsigned_type q = pv / 10000;
561 unsigned_type r = pv % 10000;
562 unsigned_type r_hi = r / 100;
563 unsigned_type r_lo = r % 100;
564 memcpy(write_pointer - 1, &internal::decimal_table[r_lo * 2], 2);
565 memcpy(write_pointer - 3, &internal::decimal_table[r_hi * 2], 2);
566 write_pointer -= 4;
567 pv = q;
568 }
569
570 // Handle remaining 1-4 digits
571 while (pv >= 100) {
572 memcpy(write_pointer - 1, &internal::decimal_table[(pv % 100) * 2], 2);
573 write_pointer -= 2;
574 pv /= 100;
575 }
576 if (pv >= 10) {
577 *write_pointer-- = char('0' + (pv % 10));
578 pv /= 10;
579 }
580 *write_pointer = char('0' + pv);
581 position += dc;
582 }
583 }
584 else SIMDJSON_IF_CONSTEXPR(std::is_floating_point<number_type>::value) {
585 constexpr size_t max_number_size = 24;
586 if (capacity_check(max_number_size)) {
587 // We could specialize for float.
588 char *end = simdjson::internal::to_chars(buffer.get() + position, nullptr,
589 double(v));
590 position = end - buffer.get();
591 }
592 }
593}
