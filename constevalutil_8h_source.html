31consteval std::string consteval_to_quoted_escaped(std::string_view input) {
32 std::string out = "\"";
33 for (char c : input) {
34 if (json_quotable_character[uint8_t(c)]) {
35 if (c == '"') {
36 out.append("\\\"");
37 } else if (c == '\\') {
38 out.append("\\\\");
39 } else {
40 std::string_view v = control_chars[uint8_t(c)];
41 out.append(v);
42 }
43 } else {
44 out.push_back(c);
45 }
46 }
47 out.push_back('"');
48 return out;
49}
