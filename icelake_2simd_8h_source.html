148 // Copies to 'output" all bytes corresponding to a 0 in the mask (interpreted as a bitset).
149 // Passing a 0 value for mask would be equivalent to writing out every byte to output.
150 // Only the first 64 - count_ones(mask) bytes of the result are significant but 64 bytes
151 // get written.
152 // Design consideration: it seems like a function with the
153 // signature simd8<L> compress(uint32_t mask) would be
154 // sensible, but the AVX ISA makes this kind of approach difficult.
155 template<typename L>
156 simdjson_inline void compress(uint64_t mask, L * output) const {
157 // we deliberately avoid _mm512_mask_compressstoreu_epi8 for portability
158 // (AMD Zen4 has terrible performance with it, it is effectively broken)
159 // _mm512_mask_compressstoreu_epi8 (output,~mask,*this);
160 __m512i compressed = _mm512_maskz_compress_epi8(~mask, *this);
161 _mm512_storeu_si512(output, compressed); // could use a mask
162 }
