15 // The allocate_padded_buffer function is a low-level function to allocate memory
16 // with padding so we can read past the "length" bytes safely. It is used by
17 // the padded_string class automatically. It returns nullptr in case
18 // of error: the caller should check for a null pointer.
19 // The length parameter is the maximum size in bytes of the string.
20 // The caller is responsible to free the memory (e.g., delete[] (...)).
21 inline char *allocate_padded_buffer(size_t length) noexcept {
22  const size_t totalpaddedlength = length + SIMDJSON_PADDING;
23  if(totalpaddedlength<length) {
24  // overflow
25  return nullptr;
26  }
27 #ifdef FUZZING_BUILD_MODE_UNSAFE_FOR_PRODUCTION
28  // avoid getting out of memory
29  if (totalpaddedlength>(1UL<<20)) {
30  return nullptr;
31  }
32 #endif
33 
34  char *padded_buffer = new (std::nothrow) char[totalpaddedlength];
35  if (padded_buffer == nullptr) {
36  return nullptr;
37  }
38  // We write nulls in the padded region to avoid having uninitialized
39  // content which may trigger warning for some sanitizers
40  std::memset(padded_buffer + length, 0, totalpaddedlength - length);
41  return padded_buffer;
42 } // allocate_padded_buffer()

125 inline simdjson_result<padded_string> padded_string::load(std::string_view filename) noexcept {
126  // Open the file
127  SIMDJSON_PUSH_DISABLE_WARNINGS
128  SIMDJSON_DISABLE_DEPRECATED_WARNING // Disable CRT_SECURE warning on MSVC: manually verified this is safe
129  std::FILE *fp = std::fopen(filename.data(), "rb");
130  SIMDJSON_POP_DISABLE_WARNINGS
131 
132  if (fp == nullptr) {
133  return IO_ERROR;
134  }
135 
136  // Get the file size
137  int ret;
138 #if SIMDJSON_VISUAL_STUDIO && !SIMDJSON_IS_32BITS
139  ret = _fseeki64(fp, 0, SEEK_END);
140 #else
141  ret = std::fseek(fp, 0, SEEK_END);
142 #endif // _WIN64
143  if(ret < 0) {
144  std::fclose(fp);
145  return IO_ERROR;
146  }
147 #if SIMDJSON_VISUAL_STUDIO && !SIMDJSON_IS_32BITS
148  __int64 llen = _ftelli64(fp);
149  if(llen == -1L) {
150  std::fclose(fp);
151  return IO_ERROR;
152  }
153 #else
154  long llen = std::ftell(fp);
155  if((llen < 0) || (llen == LONG_MAX)) {
156  std::fclose(fp);
157  return IO_ERROR;
158  }
159 #endif
160 
161  // Allocate the padded_string
162  size_t len = static_cast<size_t>(llen);
163  padded_string s(len);
164  if (s.data() == nullptr) {
165  std::fclose(fp);
166  return MEMALLOC;
167  }
168 
169  // Read the padded_string
170  std::rewind(fp);
171  size_t bytes_read = std::fread(s.data(), 1, len, fp);
172  if (std::fclose(fp) != 0 || bytes_read != len) {
173  return IO_ERROR;
174  }
175 
176  return s;
177 }
