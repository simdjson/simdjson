52 simdjson_inline simdjson_result<value> document::get_value() noexcept {
53  // Make sure we start any arrays or objects before returning, so that start_root_<object/array>()
54  // gets called.
55  iter.assert_at_document_depth();
56  switch (*iter.peek()) {
57  case '[': {
58  // The following lines check that the document ends with ].
59  auto value_iterator = get_root_value_iterator();
60  auto error = value_iterator.check_root_array();
61  if(error) { return error; }
62  return value(get_root_value_iterator());
63  }
64  case '{': {
65  // The following lines would check that the document ends with }.
66  auto value_iterator = get_root_value_iterator();
67  auto error = value_iterator.check_root_object();
68  if(error) { return error; }
69  return value(get_root_value_iterator());
70  }
71  default:
72  // Unfortunately, scalar documents are a special case in simdjson and they cannot
73  // be safely converted to value instances.
74  return SCALAR_DOCUMENT_AS_VALUE;
75  }
76 }

215 simdjson_inline simdjson_result<std::string_view> document::raw_json() noexcept {
216  auto _iter = get_root_value_iterator();
217  const uint8_t * starting_point{_iter.peek_start()};
218  auto error = consume();
219  if(error) { return error; }
220  // After 'consume()', we could be left pointing just beyond the document, but that
221  // is ok because we are not going to dereference the final pointer position, we just
222  // use it to compute the length in bytes.
223  const uint8_t * final_point{iter.unsafe_pointer()};
224  return std::string_view(reinterpret_cast<const char*>(starting_point), size_t(final_point - starting_point));
225 }

260 simdjson_inline simdjson_result<value> document::at_pointer(std::string_view json_pointer) noexcept {
261  rewind(); // Rewind the document each time at_pointer is called
262  if (json_pointer.empty()) {
263  return this->get_value();
264  }
265  json_type t;
266  SIMDJSON_TRY(type().get(t));
267  switch (t)
268  {
269  case json_type::array:
270  return (*this).get_array().at_pointer(json_pointer);
271  case json_type::object:
272  return (*this).get_object().at_pointer(json_pointer);
273  default:
274  return INVALID_JSON_POINTER;
275  }
276 }
