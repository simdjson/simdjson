124 simdjson_inline simdjson_result<size_t> value::count_elements() & noexcept {
125  simdjson_result<size_t> answer;
126  auto a = get_array();
127  answer = a.count_elements();
128  // count_elements leaves you pointing inside the array, at the first element.
129  // We need to move back so that the user can create a new array (which requires that
130  // we point at '[').
131  iter.move_at_start();
132  return answer;
133 }

204 simdjson_inline simdjson_result<value> value::at_pointer(std::string_view json_pointer) noexcept {
205  json_type t;
206  SIMDJSON_TRY(type().get(t));
207  switch (t)
208  {
209  case json_type::array:
210  return (*this).get_array().at_pointer(json_pointer);
211  case json_type::object:
212  return (*this).get_object().at_pointer(json_pointer);
213  default:
214  return INVALID_JSON_POINTER;
215  }
216 }
